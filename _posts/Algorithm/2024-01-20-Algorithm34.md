## 258. Add Digits


```c++
class Solution {
public:
    int sumDigits(int n)
    {
        int sum = 0;
        while(n >= 10)
        {
            sum += n / 10;
            n = n % 10;
        }
        return sum + n;
    }

    int addDigits(int num) {
        if(num < 10)
            return num;
        while(num >= 10)
        {
            num = sumDigits(num);
            if(num < 10)
                break;
        }
        return num;
    }
};
```
