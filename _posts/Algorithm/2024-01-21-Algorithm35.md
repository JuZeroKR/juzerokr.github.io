## 263. Ugly Number


```c++
class Solution {
public:
    bool isUgly(int n) {
        if(n==0)
            return false;
        while(true)
        {
            if(n % 2 == 0)
            {
                n /= 2;
                continue;
            }else if(n % 3 == 0)
            {
                n /= 3;
                continue;
            }else if(n % 5 == 0)
            {
                n /= 5;
                continue;
            }
            if(n==1 || n==2 || n==3 || n==5)
                break;
            return false;
        }

        return true;
    }
};
```
