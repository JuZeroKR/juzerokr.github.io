## 509. Fibonacci Number

```c++
class Solution {
private:
    int fibN[31];
public:
    int recurFib(int n) {
        if(n < 0)
            return 0;
        else if(n == 1)
            return 1;
        else if(n == 2)
            return 1;
        if(fibN[n] == -1)
        {
            int res = recurFib(n-2) + recurFib(n-1);
            fibN[n] = res;
            return res;
        }else
            return fibN[n];
    }

    int fib(int n) {
        std::fill_n(fibN, 31, -1);
        return recurFib(n);
    }
};
```

## 520. Detect Capital

```c++
class Solution {
public:
    bool isCapital(char c)
    {
        if(c >= 65 && c <= 90)
            return true;
        return false;
    }

    bool detectCapitalUse(string word) {
        int i = 0;
        for(i = 0; i < word.length(); i++)
        {
            if(isCapital(word[i]))
                continue;
            else
                break;
        }
        if(i == word.length())
            return true;

        for(i = 0; i < word.length(); i++)
        {
            if(!isCapital(word[i]))
                continue;
            else
                break;
        }
        if(i == word.length())
            return true;

        for(i = word.length() -1 ; i >= 0; i--)
        {
            if(!isCapital(word[i]))
                continue;
            else
                break;
        }
        if(i == 0)
            return true;

        return false;


    }
};
```
