## Leetcode - 226. Invert Binary Tree


```c++
class Solution {
public:
    void DoReverseTree(TreeNode* node) {
        if(node->left == NULL && node->right == NULL)
            return;
        if(node->left != NULL)
            DoReverseTree(node->left);
        if(node->right != NULL)
            DoReverseTree(node->right);
        TreeNode* temp = NULL;
        temp = node->left;
        node->left = node->right;
        node->right = temp;
    }

    TreeNode* invertTree(TreeNode* root) {
        if(root != NULL)
            DoReverseTree(root);
        return root;
    }
};

```
