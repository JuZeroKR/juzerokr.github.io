## 434. Number of Segments in a String

```c++
class Solution {
public:
    int countSegments(string s) {
        int i = 0;
        if(s.length() == 0)
            return 0;

        bool lastD;
        if(s[0] == ' ')
            lastD = false;
        else{
            lastD = true;
            i++;
        }
            
        for(auto c : s)
        {
            if(c == ' '){
                if(lastD)
                    lastD = false;
            }
            else{
                if(lastD)
                    continue;
                i++;
                lastD = true;
            }
        }
        return i;
    }
};
```


## 441. Arranging Coins
```c++
class Solution {
public:
    int arrangeCoins(int n) {
        if( n == 1 )
            return 1;
        int i = 1;
        int res = 0;
        while(i <= n)
        {
            n -= i++;
            res++;
        }
        return res;
    }
};
```
