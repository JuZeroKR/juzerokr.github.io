## Pascal's Triangle II


```c++
class Solution {
public:
    vector<int> getRow(int rowIndex) {
        
        if(rowIndex == 0)
            return vector<int> {1};
        else if(rowIndex == 1)
            return vector<int> {1,1};

        vector<int> row(rowIndex, 1);

        for(int i = 1; i< rowIndex; i++)
        {
            int last = row[0];
            for(int j = 1; j<=i ; j++)
            {
                int temp = row[j];
                row[j] = last + row[j];
                last = temp;
            }
            
        }
        row.push_back(1);
        return row;
    }
};
```
