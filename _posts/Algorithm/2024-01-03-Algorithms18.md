## 70. Climbing Stairs

```c++
class Solution {
public:
    int climbStairs(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }

        int stairs[46];
        stairs[0] = stairs[1] = 1;
        
        for (int i = 2; i <= n; i++) {
            stairs[i] = stairs[i-1] + stairs[i-2];
        }
        return stairs[n];
    }
};
```
