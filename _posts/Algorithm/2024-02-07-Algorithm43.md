## 389. Find the Difference

```c++
class Solution {
public:
    char findTheDifference(string s, string t) {
        int characterS[26] = {0, };
        int characterT[26] = {0, };
        for(int i = 0; i < s.length(); i++)
        {
            int indexS = s[i] - 97;
            int indexT = t[i] - 97;
            characterS[indexS]++;
            characterT[indexT]++;
        }
        characterT[t[t.length()-1] - 97]++;

        int i = 0;
        for(; i<26; i++)
        {
            if(characterS[i] != characterT[i])
                break;
        }
        return i + 97;

    }
};
```
