## 66. Plus One


```c++
class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
        int digitsLength = digits.size()-1;
        int i = 0;
        int carry = 1;
        int cur = 0;
        vector<int> dig;
        while(digitsLength != -1)
        {
            int cur = digits[digitsLength--];
            int sum = cur+carry;
            if(sum >= 10){
                carry = 1;
                sum %= 10;
            }
            else{
                carry = 0;
            }
            dig.insert(dig.begin(), sum);
        }
        if(carry == 1)
            dig.insert(dig.begin(), carry);
        return dig;
    }
};
```
