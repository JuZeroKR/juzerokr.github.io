## 412. Fizz Buzz

```c++
class Solution {
public:
    vector<string> fizzBuzz(int n) {
        vector<string> s;
        for(int i=1 ; i<=n ; i++)
        {
            string str = "";
            if(i % 3 == 0 && i % 5 == 0)
            {
                str = "FizzBuzz";
            }else if(i%3 == 0)
            {
                str = "Fizz";
            }else if(i%5 == 0)
            {
                str = "Buzz";
            }else
            {
                str = to_string(i);
            }
            s.push_back(str);
        }
        return s;
    }
};
```


## 414. Third Maximum Number

```c++
class Solution {
public:
    int thirdMax(vector<int>& nums) {
        vector<int> thirdMax;
        set<int> setNums;
        for(int i = 0; i < nums.size(); i++)
        {
            setNums.insert(nums[i]);
        }

        for(auto n : setNums)
        {
            for(int j = 0; j < 3 ; j++)
            {
                if(thirdMax.size() < j+1)
                    break;
                if(thirdMax[j] < n)
                {
                    thirdMax.insert(thirdMax.begin(), n);
                    break;
                }else if(thirdMax[j] == n)
                {
                    break;
                }
            }
            if(thirdMax.size() > 3)
                thirdMax.erase(thirdMax.begin()+3, thirdMax.begin()+thirdMax.size());
            else
                thirdMax.push_back(n);
        }
        return thirdMax.back();
    }
};
```
