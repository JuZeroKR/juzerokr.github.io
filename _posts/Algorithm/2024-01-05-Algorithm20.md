## 27. Remove Element


```c++
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int j = 0;
        for(int i = 0;i<nums.size(); i++){
            if(nums[i] != val){
                nums[j++] = nums[i];
            }
        }
        return j;
    }
};
```


## 26. Remove Duplicates from Sorted Array


```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int i = 0;
        int j = 1;
        if(nums.size() == 1)
            return j;
        while(i < nums.size()-1)
        {
            if(nums[i] != nums[i+1])
                nums[j++] = nums[i+1];
            i++;
        }
        return j;
    }
};
```
