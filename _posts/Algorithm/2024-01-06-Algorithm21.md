## 80. Remove Duplicates from Sorted Array II


```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int i = 0;
        int j = 0;
        int left = 0;
        int right = 0;
        if(nums.size() == 1)
            return 1;
        while(right < nums.size())
        {
            if(nums[left] != nums[right])
            {
                if(right - left > 1)
                {
                    nums[j++] = nums[left];
                    nums[j++] = nums[left];
                }else if(right-left == 1)
                {
                    nums[j++] = nums[left];
                }
                left = right;
            }
            right++;
        }
        if(right - left > 1)
        {
            nums[j++] = nums[left];
            nums[j++] = nums[left];
        }else if(right-left == 1)
        {
            nums[j++] = nums[left];
        }
        return j;
    }
};
```
