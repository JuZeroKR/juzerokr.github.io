## 463. Island Perimeter

```c++
class Solution {
public:
    int islandPerimeter(vector<vector<int>>& grid) {
        // 경계면이면 +1
        // 바다면 +1
        int boardNum = 0;
        // 0, grid.size()-1
        for(int i = 0; i < grid.size(); i++)
        {
            for(int j = 0; j < grid[i].size(); j++)
            {
                // 바다면 continue
                int isGround = grid[i][j];
                if(isGround == 0)
                    continue;

                int blockBoardNum = 0;;
                // 경계면인지 확인
                for(int t = 0; t < 4; t++)
                {
                    int x = i;
                    int y = j;
                    
                    if(t == 0)
                        x--;
                    else if(t==1)
                        x++;
                    else if(t==2)
                        y--;
                    else
                        y++;
                    if(x < 0 || x >= grid.size() || y < 0 || y >= grid[i].size()){
                        blockBoardNum++;
                        continue;
                    }
                    if(grid[x][y] == 0){
                        blockBoardNum++;
                    }
                        
                }
                boardNum += blockBoardNum; 
            }
        }
        return boardNum;
    }
};
```

## 476. Number Complement

```c++
class Solution {
public:
    int findComplement(int num) {
        int units = 0;
        int n = num;
        while(n != 0)
        {
            n /= 2;
            units++;
        }
        int base = (int)(pow(2,units)-1);
        return num ^ base;
    }
};
```
