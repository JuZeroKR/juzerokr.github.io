## 231. Power of Two


```c++
class Solution {
public:
    bool isPowerOfTwo(int n) {
        if(n < 1 || (n > 1 && n%2 == 1))
            return false;
        int compareN = n-1 & n;
        if(compareN != 0)
            return false;
        return true;
    }
};
```


## 232. Implement Queue using Stacks


```c++
class MyStack {
public:
    queue<int> q; 
    MyStack() {
        
    }
    
    void push(int x) {
        q.push(x);
    }
    
    int pop() {
        queue<int> tmp;
        while(q.size() > 1)
        {
            tmp.push(q.front());
            q.pop();
        }
        swap(q, tmp);
        return tmp.back();
    }
    
    int top() {
        return q.back();
    }
    
    bool empty() {
        return q.empty();
    }
};
```
