## 67. Add Binary


```c++
class Solution {
public:
    string calc(char a_i, char b_i, int& carry){
        int a = 0;
        int b = 0;
        if(a_i == '1')
            a = 1;
        if(b_i == '1')
            b = 1;

        int sum = a + b + carry;

        if(sum == 3){
            carry = 1;
            return "1";
        }else if(sum == 2){
            carry = 1;
            return "0";
        }else if(sum == 1){
            carry = 0;
            return "1";
        }
        carry = 0;
        return "0";
    }

    string addBinary(string a, string b) {
        int padding = 0;
        string output = "";
        if(a.length() > b.length()){
            padding = a.length() - b.length();
            for(int i=0;i<padding; i++)
                b.insert(0, "0");
        }else{
            padding = b.length() - a.length();
            for(int i=0;i<padding; i++)
                a.insert(0, "0");
        }
        int carry = 0;
        for(int i = a.length()-1 ; i>=0 ; i--)
        {
            output.insert(0, calc(a[i], b[i], carry));
        }
        if(carry == 1)
            output.insert(0, "1");
        return output;

    }
};
```
